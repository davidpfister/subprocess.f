204 function process_new(path, stdin, stdout, stderr) result(that)
205 character(*), intent(in) :: path
206 procedure(process_io), intent(out), pointer, optional :: stdin
207 procedure(process_io), optional :: stdout
208 procedure(process_io), optional :: stderr
209 type(process) :: that
210
211 call internal_finalize(that%ptr)
212
213 that%is_running = .false.
214 that%path = trim(path)
215 if (present(stdin)) then
216 nullify(stdin)
217 stdin => process_writeto_stdin
218 end if
219 if (present(stdout)) that%stdout => stdout
220 if (present(stderr)) that%stderr => stderr
221 end function

354 class(process), intent(inout) :: this
355 type(string), intent(in) :: args(:)
356 !private
357 character(:), allocatable :: cmd, arg
358 procedure(process_io), pointer :: fptr => null()
359 integer :: i
360
361 cmd = this%path

363 cmd = trim(cmd)//' '//trim(args(i))
364 end do

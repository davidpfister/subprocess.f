502 function process_new(path, stdin, stdout, stderr) result(that)
503 character(*), intent(in) :: path
504 procedure(process_io), intent(out), pointer, optional :: stdin
505 procedure(process_io), optional :: stdout
506 procedure(process_io), optional :: stderr
507 type(process) :: that
508
509 call internal_finalize(that%ptr)
510
511 that%is_running = .false.
512 that%path = trim(path)
513 if (present(stdin)) then
514 nullify(stdin)
515 stdin => process_writeto_stdin
516 end if
517 if (present(stdout)) that%stdout => stdout
518 if (present(stderr)) that%stderr => stderr
519 end function

664 subroutine process_run_with_args(this, args, option)
665 class(process), intent(inout) :: this
666 type(string), intent(in) :: args(:)
667 integer(option_enum), intent(in), optional :: option
668 !private
669 character(:), allocatable :: cmd, arg
670 procedure(process_io), pointer :: fptr => null()
671 integer :: i
672
673 cmd = this%path
674 do i = 1, size(args)
675 cmd = trim(cmd)//' '//trim(args(i))
676 end do

678 if (allocated(this%excode)) deallocate(this%excode)
679 allocate(this%excode, source = 0)
680
681 this%is_running = .true.
682
683 call get_time(this%begtime)
684 if (present(option)) then
685 this%pid = internal_run(cmd, option, this%ptr, this%excode)

688 end if

692 if (associated(this%stdout)) then

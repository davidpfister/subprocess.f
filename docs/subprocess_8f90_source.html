504 function process_new(path, stdin, stdout, stderr) result(that)
505 character(*), intent(in) :: path
506 procedure(process_io), intent(out), pointer, optional :: stdin
507 procedure(process_io), optional :: stdout
508 procedure(process_io), optional :: stderr
509 type(process) :: that
510
511 call internal_finalize(that%ptr)
512
513 that%is_running = .false.
514 that%path = trim(path)
515 that%isasync = .false.
516 if (present(stdin)) then
517 nullify(stdin)
518 stdin => process_writeto_stdin
519 end if
520 if (present(stdout)) that%stdout => stdout
521 if (present(stderr)) that%stderr => stderr
522 end function

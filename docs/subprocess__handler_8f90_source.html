209 integer function internal_run_default(cmd, fp, excode) result(pid)
210 character(*), intent(in) :: cmd
211 type(handle_pointer), intent(inout) :: fp
212 integer(c_int), intent(out) :: excode
213 !private
214 integer :: ierr
215
216 if (.not. allocated(fp%handle)) allocate(fp%handle)
217 pid = subprocess_create_c(cmd // c_null_char, option_inherit_environment, fp%handle)
218 if (pid < 0) then
219 write (*, *) '*run* ERROR: Could not create process!'
220 excode = -1
221 end if
222
223 ierr = subprocess_join_c(fp%handle, excode)
224 end function

234 integer function internal_run_with_options(cmd, options, fp, excode) result(pid)
235 character(*), intent(in) :: cmd
236 integer(option_enum), intent(in) :: options(..)
237 type(handle_pointer), intent(inout) :: fp
238 integer(c_int), intent(out) :: excode
239 !private
240 integer :: ierr, i
241 integer(option_enum) :: opt
242
243 opt = option_inherit_environment
244 select rank(o => options)
245 rank(0)
246 opt = merge(ior(opt, o), option_none, o /= option_none)
247 rank(1)
248 do i = 1, size(o)
249 if (o(i) == option_none) then
250 opt = option_none
251 exit
252 end if
253 opt = ior(opt, o(i))
254 end do
255 rank default
256 opt = option_inherit_environment
257 end select
258
259 if (.not. allocated(fp%handle)) allocate(fp%handle)
260 pid = subprocess_create_c(cmd // c_null_char, opt, fp%handle)
261 if (pid < 0) then
262 write (*, *) '*run* ERROR: Could not create process!'

354 type(handle_pointer), intent(inout) :: fp

364 integer :: n
365
366 if (allocated(fp%handle)) then
